https://massgrave.dev/

# comando para conseguir la product key de Windows
Get-ItemProperty -Path "HKLM:\SOFTWARE\Microsoft\Windows NT\CurrentVersion\SoftwareProtectionPlatform" -Name "BackupProductKeyDefault"
  # el anterior comando equivale a la siguiente busqueda
  > regedit
  > Equipo\HKEY_LOCAL_MACHINE\SOFTWARE\Microsoft\Windows NT\CurrentVersion\SoftwareProtectionPlatform 
    > BackupProductKeyDefault
